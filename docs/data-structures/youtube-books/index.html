    result.add(current); //r["ad","ae", "af"]

    letterCombinationsRecursive(result, digits, current + letters.charAt(i), index+1, mapping);

    }

    String s = "226"; // BZ, BBF, VF => 3 String s = "111"; // AAA, AK, KA => 3

    1. Dynamic Programming - Bottom up approach
    2. dp[] = new int[s.lenghth()+1] to consider the whole String
    3. Base case: 0=1; 1=if '0' 0 not 1
    4. Starting from 2; check for decode ways for first and second digit
    5. return the index of string length in dp array

    int numDecodings(String s) { int[] dp = new int[s.length()+1]; dp[0] = 1; dp[1] = s.charAt(0) != '0' ? 1 : 0; for (int i = 2; i <= s.length(); i++) { int first = Integer.valueOf(s.substring(i-1, i)); int second = Integer.valueOf(s.substring(i-2, i)); if (first >= 1) { dp[i] += dp[i-1]; } if (second >= 10 && second <= 26) { dp[i] += dp[i-2]; } } return dp[s.length()]; }

  • BINARY TREE PATHS

    1. DFS dfs(Node root, String path, List<String> paths)
    2. Since root is not null, add root value to path; path += root.val;
    3. If you reach the leaf node, add the path to the result list if (root.left == null && root.right == null) { paths.add(path); }
    4. If the root has a left child, call dfs with root's left dfs(root.left, path + "->", paths)
    5. If the root has a right child, call dfs with root's right dfs(root.right, path + "->", paths)
    6. return result list paths

  • REORGANIZE STRING

    1. Create a Map to count the number of occurances of each character for (char c: s.toCharArray()) { map.put(c, map.getOrDefault(c, 0) + 1}
    2. Create a PriorityQueue maxHeap sorted by occurances in map PQ<Character> maxHeap=new PQ((a,b)-> map.get(b) - map.get(a))
    3. Add all keys to PQ maxHeap.addAll(map.keySet())
    4. while (maxHeap.size() > 1)
    5. remove top elements from PQ char current = maxHeap.remove(); char next = maxHeap.remove();
    6. Add to result sb.append(current); sb.append(next);
    7. Reduce the count in the map map.put(current, map.get(current) - 1) map.put(next, map.get(next) - 1)
    8. If there are any remaining, add them back to maxHeap if (map.get(current) > 0) maxHeap.add(current) if (map.get(next) > 0) maxHeap.add(next)
    9. End while
    10. After processing if PQ is not empty and if there is ony one char left, add it to the result, else return empty string if (!maxHeap.isEmpty()) { char last = maxHeap.remove(); if (map.get(last) > 1) { return ""; } else { sb.append(last); } }
    11. return result. sb.toString();
